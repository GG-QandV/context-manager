import { describe, it, mock, beforeEach, afterEach } from 'node:test';
import assert from 'node:assert';

describe('runAudit', () => {
  beforeEach(() => {
    process.env.DATABASE_URL = 'postgresql://postgres:pass@localhost:5432/test';
  });

  afterEach(() => {
    delete process.env.DATABASE_URL;
    mock.restoreAll();
  });

  async function freshAudit() {
    delete require.cache[require.resolve('../src/services/audit.service')];
    const mod = await import('../src/services/audit.service');
    return mod.runAudit;
  }

  it('returns no discrepancies when PG and Qdrant match', async () => {
    const { postgresService } = require('../src/services/postgres.service');
    const { qdrantService } = require('../src/services/qdrant.service');

    mock.method(postgresService, 'getAllSyncedSyncIds', async () => ['a', 'b', 'c']);
    mock.method(qdrantService, 'getAllDistinctSyncIds', async () => ['a', 'b', 'c']);

    const runAudit = await freshAudit();
    const result = await runAudit();

    assert.deepStrictEqual(result.missingInQdrant, []);
    assert.deepStrictEqual(result.missingInPg, []);
  });

  it('detects records missing in Qdrant', async () => {
    const { postgresService } = require('../src/services/postgres.service');
    const { qdrantService } = require('../src/services/qdrant.service');

    mock.method(postgresService, 'getAllSyncedSyncIds', async () => ['a', 'b', 'c', 'd']);
    mock.method(qdrantService, 'getAllDistinctSyncIds', async () => ['a', 'b']);

    const runAudit = await freshAudit();
    const result = await runAudit();

    assert.deepStrictEqual(result.missingInQdrant, ['c', 'd']);
    assert.deepStrictEqual(result.missingInPg, []);
  });

  it('detects orphaned points in Qdrant (missing in PG)', async () => {
    const { postgresService } = require('../src/services/postgres.service');
    const { qdrantService } = require('../src/services/qdrant.service');

    mock.method(postgresService, 'getAllSyncedSyncIds', async () => ['a']);
    mock.method(qdrantService, 'getAllDistinctSyncIds', async () => ['a', 'orphan-1', 'orphan-2']);

    const runAudit = await freshAudit();
    const result = await runAudit();

    assert.deepStrictEqual(result.missingInPg, ['orphan-1', 'orphan-2']);
  });

  it('handles empty PG and empty Qdrant', async () => {
    const { postgresService } = require('../src/services/postgres.service');
    const { qdrantService } = require('../src/services/qdrant.service');

    mock.method(postgresService, 'getAllSyncedSyncIds', async () => []);
    mock.method(qdrantService, 'getAllDistinctSyncIds', async () => []);

    const runAudit = await freshAudit();
    const result = await runAudit();

    assert.strictEqual(result.syncedCount, 0);
    assert.strictEqual(result.missingInQdrant.length, 0);
    assert.strictEqual(result.missingInPg.length, 0);
  });
});
